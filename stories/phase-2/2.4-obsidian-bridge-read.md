## Context

Doug's Obsidian vault at `/Users/doug/Obsidian` is the source of truth for tasks and notes. ARIA reads this vault via two methods:
1. **Direct filesystem** (Node.js `fs` module) — for reading markdown files directly in the Next.js API route (works on local Mac dev)
2. **Obsidian Local REST API plugin** — available if the plugin is installed and Obsidian is open (provides richer metadata)

This story implements filesystem-based reads. The Local REST API is used as a bonus if the plugin is available.

## Technical Instructions

### Step 1: Create lib/obsidian.ts

```typescript
import fs from 'fs';
import path from 'path';
import matter from 'gray-matter';
import type { ObsidianTask, ObsidianNote } from '@/types';
import { config } from '@/lib/config';

const VAULT = config.obsidianVaultPath;

// ---- Tasks ----

export function readAllTasks(): ObsidianTask[] {
  const markdownFiles = getAllMarkdownFiles(VAULT);
  const tasks: ObsidianTask[] = [];

  for (const filePath of markdownFiles) {
    const content = fs.readFileSync(filePath, 'utf-8');
    const lines = content.split('\n');

    lines.forEach((line, lineNumber) => {
      // Match Obsidian task syntax: - [ ] task text
      const taskMatch = line.match(/^(\s*)-\s+\[([ x])\]\s+(.+)$/);
      if (!taskMatch) return;

      const [, , checkmark, text] = taskMatch;
      const completed = checkmark === 'x';

      // Extract due date if present: 📅 2024-04-15
      const dueDateMatch = text.match(/📅\s*(\d{4}-\d{2}-\d{2})/);
      const dueDate = dueDateMatch?.[1];

      // Extract tags: #tag
      const tags = [...text.matchAll(/#([\w-]+)/g)].map(m => m[1]);

      tasks.push({
        id: `${filePath}:${lineNumber}`,
        text: text.replace(/📅\s*\d{4}-\d{2}-\d{2}/g, '').replace(/#[\w-]+/g, '').trim(),
        completed,
        filePath: path.relative(VAULT, filePath),
        lineNumber,
        dueDate,
        tags,
      });
    });
  }

  // Return only open tasks, sorted by due date
  return tasks
    .filter(t => !t.completed)
    .sort((a, b) => {
      if (a.dueDate && b.dueDate) return a.dueDate.localeCompare(b.dueDate);
      if (a.dueDate) return -1;
      if (b.dueDate) return 1;
      return 0;
    });
}

// ---- Notes ----

export function readRecentNotes(limit = 10): ObsidianNote[] {
  const markdownFiles = getAllMarkdownFiles(VAULT);

  return markdownFiles
    .map(filePath => {
      const stats = fs.statSync(filePath);
      const content = fs.readFileSync(filePath, 'utf-8');
      const { data: frontmatter, content: body } = matter(content);
      const relativePath = path.relative(VAULT, filePath);
      const folder = path.dirname(relativePath);

      return {
        path: relativePath,
        title: frontmatter.title ?? path.basename(filePath, '.md'),
        content: body.slice(0, 500), // first 500 chars as preview
        folder,
        created: frontmatter.created ?? stats.birthtime.toISOString(),
        modified: stats.mtime.toISOString(),
        tags: frontmatter.tags ?? [],
      } satisfies ObsidianNote;
    })
    .sort((a, b) => b.modified.localeCompare(a.modified))
    .slice(0, limit);
}

export function searchNotes(query: string, limit = 20): ObsidianNote[] {
  const markdownFiles = getAllMarkdownFiles(VAULT);
  const lowerQuery = query.toLowerCase();

  return markdownFiles
    .filter(filePath => {
      const content = fs.readFileSync(filePath, 'utf-8').toLowerCase();
      return content.includes(lowerQuery);
    })
    .map(filePath => {
      const stats = fs.statSync(filePath);
      const content = fs.readFileSync(filePath, 'utf-8');
      const { data: frontmatter, content: body } = matter(content);
      const relativePath = path.relative(VAULT, filePath);

      return {
        path: relativePath,
        title: frontmatter.title ?? path.basename(filePath, '.md'),
        content: body.slice(0, 500),
        folder: path.dirname(relativePath),
        created: frontmatter.created ?? stats.birthtime.toISOString(),
        modified: stats.mtime.toISOString(),
        tags: frontmatter.tags ?? [],
      } satisfies ObsidianNote;
    })
    .slice(0, limit);
}

// ---- Helpers ----

function getAllMarkdownFiles(dir: string): string[] {
  const results: string[] = [];

  if (!fs.existsSync(dir)) return results;

  const entries = fs.readdirSync(dir, { withFileTypes: true });
  for (const entry of entries) {
    // Skip hidden directories (like .obsidian, .trash)
    if (entry.name.startsWith('.')) continue;

    const fullPath = path.join(dir, entry.name);
    if (entry.isDirectory()) {
      results.push(...getAllMarkdownFiles(fullPath));
    } else if (entry.name.endsWith('.md')) {
      results.push(fullPath);
    }
  }

  return results;
}
```

### Step 3: Create tasks API route

Create `app/api/tasks/route.ts`:

```typescript
import { NextResponse } from 'next/server';
import { auth } from '@/lib/auth';
import { readAllTasks } from '@/lib/obsidian';
import { config } from '@/lib/config';

export async function GET() {
  const session = await auth();
  if (!session) return NextResponse.json({ error: 'Unauthorized' }, { status: 401 });

  if (config.demoMode) {
    return NextResponse.json({ tasks: getMockTasks() });
  }

  try {
    const tasks = readAllTasks();
    return NextResponse.json({ tasks });
  } catch (error) {
    console.error('Obsidian tasks error:', error);
    return NextResponse.json({ error: 'Failed to read vault tasks' }, { status: 500 });
  }
}

function getMockTasks() {
  return [
    { id: 'mock-task-1', text: 'Review ARIA architecture doc', completed: false, filePath: 'Tasks/Work.md', lineNumber: 5, dueDate: new Date().toISOString().split('T')[0] },
    { id: 'mock-task-2', text: 'Submit homework assignment', completed: false, filePath: 'Tasks/School.md', lineNumber: 3, dueDate: null },
    { id: 'mock-task-3', text: 'Grocery shopping', completed: false, filePath: 'Tasks/Personal.md', lineNumber: 1, dueDate: null },
  ];
}
```

### Step 4: Create notes API route

Create `app/api/notes/route.ts`:

```typescript
import { NextRequest, NextResponse } from 'next/server';
import { auth } from '@/lib/auth';
import { readRecentNotes, searchNotes } from '@/lib/obsidian';

export async function GET(request: NextRequest) {
  const session = await auth();
  if (!session) return NextResponse.json({ error: 'Unauthorized' }, { status: 401 });

  const { searchParams } = new URL(request.url);
  const query = searchParams.get('q');

  try {
    const notes = query ? searchNotes(query) : readRecentNotes(10);
    return NextResponse.json({ notes });
  } catch (error) {
    console.error('Obsidian notes error:', error);
    return NextResponse.json({ error: 'Failed to read vault notes' }, { status: 500 });
  }
}
```

## Notes for Claude Code

- Node.js `fs` module is available in Next.js API routes (server-side only) — never import it in client components
- The vault path must be set in `.env.local` as `OBSIDIAN_VAULT_PATH=/Users/doug/Obsidian`
- Task syntax follows the Obsidian convention: `- [ ] task` for open, `- [x] task` for done
- `gray-matter` parses YAML frontmatter from markdown files — this is how Obsidian stores metadata
- Don't try to parse complex wiki-links or backlinks yet — that's a stretch feature
