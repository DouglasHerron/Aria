## Technical Instructions

### Step 1: Add write functions to lib/obsidian.ts

```typescript
import fs from 'fs';
import path from 'path';
import { config } from '@/lib/config';

const VAULT = config.obsidianVaultPath;

// Mark a task as complete by updating its checkbox in the file
export function markTaskComplete(filePath: string, lineNumber: number): void {
  const absolutePath = path.join(VAULT, filePath);
  const content = fs.readFileSync(absolutePath, 'utf-8');
  const lines = content.split('\n');
  
  const line = lines[lineNumber];
  if (!line) throw new Error(`Line ${lineNumber} not found in ${filePath}`);
  
  // Replace - [ ] with - [x] and add completion date
  const today = new Date().toISOString().split('T')[0];
  lines[lineNumber] = line
    .replace(/- \[ \]/, `- [x]`)
    .trimEnd() + ` ✅ ${today}`;
  
  fs.writeFileSync(absolutePath, lines.join('\n'), 'utf-8');
}

// Create a new task in the specified vault file (default: Tasks/Inbox.md)
export function createTask(
  text: string,
  options: {
    targetFile?: string; // relative to vault, e.g. 'Tasks/Work.md'
    dueDate?: string;    // YYYY-MM-DD
    tags?: string[];
  } = {}
): { filePath: string; lineNumber: number } {
  const targetFile = options.targetFile ?? 'Tasks/Inbox.md';
  const absolutePath = path.join(VAULT, targetFile);
  
  // Create directory and file if they don't exist
  fs.mkdirSync(path.dirname(absolutePath), { recursive: true });
  if (!fs.existsSync(absolutePath)) {
    fs.writeFileSync(absolutePath, `# Inbox\n\n`, 'utf-8');
  }
  
  // Build task line
  const duePart = options.dueDate ? ` 📅 ${options.dueDate}` : '';
  const tagPart = options.tags?.length ? ' ' + options.tags.map(t => `#${t}`).join(' ') : '';
  const taskLine = `- [ ] ${text}${duePart}${tagPart}`;
  
  const content = fs.readFileSync(absolutePath, 'utf-8');
  const lines = content.split('\n');
  lines.push(taskLine);
  fs.writeFileSync(absolutePath, lines.join('\n'), 'utf-8');
  
  return { filePath: targetFile, lineNumber: lines.length - 1 };
}

// Create a new note in the vault
export function createNote(
  title: string,
  content: string,
  options: {
    folder?: string; // relative to vault, e.g. 'Notes/Work'
    tags?: string[];
  } = {}
): { path: string } {
  const folder = options.folder ?? 'Notes';
  const filename = `${title.replace(/[^\w\s-]/g, '').trim().replace(/\s+/g, '-')}.md`;
  const relativePath = path.join(folder, filename);
  const absolutePath = path.join(VAULT, relativePath);
  
  fs.mkdirSync(path.dirname(absolutePath), { recursive: true });
  
  const today = new Date().toISOString().split('T')[0];
  const frontmatter = [
    '---',
    `title: ${title}`,
    `created: ${today}`,
    `tags: [${(options.tags ?? []).join(', ')}]`,
    '---',
    '',
  ].join('\n');
  
  fs.writeFileSync(absolutePath, frontmatter + content, 'utf-8');
  
  return { path: relativePath };
}
```

### Step 2: Update tasks API route

In `app/api/tasks/route.ts`, add POST (create) and PATCH (complete):

```typescript
// POST /api/tasks — create a new task
export async function POST(request: NextRequest) {
  const session = await auth();
  if (!session) return NextResponse.json({ error: 'Unauthorized' }, { status: 401 });

  const { text, targetFile, dueDate, tags } = await request.json();
  if (!text) return NextResponse.json({ error: 'text is required' }, { status: 400 });

  try {
    const result = createTask(text, { targetFile, dueDate, tags });
    return NextResponse.json({ success: true, ...result });
  } catch (error) {
    return NextResponse.json({ error: String(error) }, { status: 500 });
  }
}

// PATCH /api/tasks — mark a task complete
export async function PATCH(request: NextRequest) {
  const session = await auth();
  if (!session) return NextResponse.json({ error: 'Unauthorized' }, { status: 401 });

  const { filePath, lineNumber } = await request.json();
  if (!filePath || lineNumber === undefined) {
    return NextResponse.json({ error: 'filePath and lineNumber required' }, { status: 400 });
  }

  try {
    markTaskComplete(filePath, lineNumber);
    return NextResponse.json({ success: true });
  } catch (error) {
    return NextResponse.json({ error: String(error) }, { status: 500 });
  }
}
```

```typescript
// POST /api/notes — create a new note
export async function POST(request: NextRequest) {
  const session = await auth();
  if (!session) return NextResponse.json({ error: 'Unauthorized' }, { status: 401 });

  const { title, content, folder, tags } = await request.json();
  if (!title) return NextResponse.json({ error: 'title is required' }, { status: 400 });

  try {
    const result = createNote(title, content ?? '', { folder, tags });
    return NextResponse.json({ success: true, ...result });
  } catch (error) {
    return NextResponse.json({ error: String(error) }, { status: 500 });
  }
}
```

## Notes for Claude Code

- File writes are synchronous (fs.writeFileSync) — this is fine for a single-user personal tool
- `path.join(VAULT, filePath)` is critical to prevent path traversal — always join with the vault root
- The task `id` format is `filePath:lineNumber` — parse this when calling PATCH
- After a task is marked complete, remove it from the widget list optimistically (don't wait for a full refetch)
- Obsidian will pick up file changes automatically via its file watcher — no Obsidian restart needed
