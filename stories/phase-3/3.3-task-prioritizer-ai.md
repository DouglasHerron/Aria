## Technical Instructions

### Step 1: Create lib/ai/prompts/tasks.ts

```typescript
import { ai } from '@/lib/ai';
import type { ObsidianTask } from '@/types';

const TASK_SYSTEM_PROMPT = `You are ARIA's task prioritization engine. Analyze a list of open tasks and return structured JSON.

Prioritization criteria (in order of importance):
1. Due date: overdue tasks are critical; due today are high; due this week are medium
2. Task text keywords: words like "urgent", "ASAP", "deadline", "submit", "review" indicate high priority
3. Context: work tasks during business hours, school tasks before deadlines

Always return valid JSON with this structure:
{
  "top3": ["task-id-1", "task-id-2", "task-id-3"],
  "scores": [
    { "id": "task-id", "score": 0-100, "reason": "Brief reason for this score" }
  ],
  "focusSuggestion": "One sentence about what Doug should tackle first today and why"
}

Score 0-100: 90-100 = critical/overdue, 70-89 = high, 40-69 = medium, 0-39 = low
top3: The IDs of the 3 highest priority tasks Doug should focus on today`;

export interface TaskPrioritizationResult {
  top3: string[];
  scores: Array<{ id: string; score: number; reason: string }>;
  focusSuggestion: string;
}

export async function prioritizeTasks(
  tasks: ObsidianTask[]
): Promise<TaskPrioritizationResult> {
  if (tasks.length === 0) {
    return {
      top3: [],
      scores: [],
      focusSuggestion: 'No open tasks. Consider reviewing your goals or planning ahead.',
    };
  }

  const today = new Date().toISOString().split('T')[0];

  const prompt = `Today is ${today}. Prioritize these ${tasks.length} open tasks:

${tasks.map(t => `ID: ${t.id}
Text: ${t.text}
Due: ${t.dueDate ?? 'No due date'}
File: ${t.filePath}
Tags: ${t.tags?.join(', ') ?? 'none'}
`).join('\n')}`;

  const result = await ai.complete({
    system: TASK_SYSTEM_PROMPT,
    prompt,
    maxTokens: 768,
  });

  try {
    const json = result.text.replace(/```json\n?|\n?```/g, '').trim();
    const parsed = JSON.parse(json);
    return parsed;
  } catch {
    // Fallback: use due date ordering
    const sorted = [...tasks].sort((a, b) => {
      if (a.dueDate && b.dueDate) return a.dueDate.localeCompare(b.dueDate);
      if (a.dueDate) return -1;
      if (b.dueDate) return 1;
      return 0;
    });
    return {
      top3: sorted.slice(0, 3).map(t => t.id),
      scores: tasks.map(t => ({ id: t.id, score: 50, reason: 'Default priority' })),
      focusSuggestion: 'Focus on your earliest due date tasks first.',
    };
  }
}
```

### Step 2: Update tasks API route

In `app/api/tasks/route.ts`, update GET to add AI prioritization:

```typescript
import { readAllTasks } from '@/lib/obsidian';
import { prioritizeTasks } from '@/lib/ai/prompts/tasks';

// In GET handler:
const tasks = readAllTasks();
const prioritization = await prioritizeTasks(tasks).catch(() => null);

// Merge AI scores into tasks
const enrichedTasks = tasks.map((task, index) => {
  const score = prioritization?.scores.find(s => s.id === task.id);
  const rank = prioritization?.top3.indexOf(task.id);
  return {
    ...task,
    priorityScore: score?.score,
    focusRank: rank !== undefined && rank >= 0 ? rank + 1 : undefined,
  };
});

return NextResponse.json({
  tasks: enrichedTasks,
  top3Ids: prioritization?.top3 ?? [],
  focusSuggestion: prioritization?.focusSuggestion ?? null,
});
```
